  // ── Dynamic savings calculation ──
  // Call this when the user selects a treatment in your booking form
  function updateUpsellPricing(singlePrice, memberPrice, deliveryFee) {
    const savings = singlePrice - memberPrice + deliveryFee;

    // Update toast
    const toast = document.getElementById('inlineToast');
    if (toast) {
      toast.querySelector('.toast-savings-badge').textContent = `SAVE $${singlePrice - memberPrice}`;
      toast.querySelector('.toast-text').innerHTML =
        `<strong>Members pay $${memberPrice}</strong> for this session instead of $${singlePrice}`;
    }

    // Update full card
    const card = document.getElementById('upsellCard');
    if (card) {
      card.querySelector('.upsell-banner-text').innerHTML =
        `You'd <strong>save $${savings} today</strong> with a membership`;
      card.querySelector('.price-option.single .option-price').textContent = `$${singlePrice}`;
      card.querySelector('.price-option.member .option-price').textContent = `$${memberPrice}`;
      card.querySelector('.price-option.member .option-detail').textContent = `Save $${singlePrice - memberPrice}/session`;
    }
  }

  // ── Dismiss/restore upsell ──
  function dismissUpsell() {
    document.getElementById('upsellCard').classList.add('dismissed');
    setTimeout(() => {
      document.getElementById('upsellMini').classList.add('visible');
    }, 400);
  }

  function restoreUpsell() {
    document.getElementById('upsellMini').classList.remove('visible');
    const card = document.getElementById('upsellCard');
    card.classList.remove('dismissed');
    card.scrollIntoView({ behavior: 'smooth', block: 'center' });
  }
